#!/bin/bash

# Accept Name and Roll via positional arguments
name="$1"
roll="$2"

# If name is empty → prompt interactively
if [ -z "$name" ]; then
    read -p "Enter Name: " name
fi

# Trim whitespace
name="$(echo "$name" | xargs)"

# If roll is empty → prompt interactively
if [ -z "$roll" ]; then
    read -p "Enter Roll: " roll
fi

# Validate roll number (must be numeric)
if ! [[ "$roll" =~ ^[0-9]+$ ]]; then
    echo "ERROR: Roll number must be numeric" >&2
    exit 1
fi

# Print output in required format
echo "Student: $name"
echo "Roll: $roll"
